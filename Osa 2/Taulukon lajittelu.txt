#include <iostream>
#include <typeinfo>

using namespace std;

void lajittele(T &taulukko, size_t N){
    cout << "Taulukossa on " << N << " alkiota" << endl;
    cout << "Taulukon tyyppi-info: " << typeid(taulukko).name() << endl;
    cout << "Taulukon muistiosoite: " << &taulukko << endl;
}

int main(int argc, char *argv[])
{
    int kokonaislukuTaulu[5] = {4,7,66,53,2};
    int taulukonKoko = sizeof(kokonaislukuTaulu) / sizeof(kokonaislukuTaulu[0]);
    lajittele(kokonaislukuTaulu, taulukonKoko);
    char merkkiTaulu[5] = {'y', 'F', 'z'};
    taulukonKoko = sizeof(merkkiTaulu) / sizeof(merkkiTaulu[0]);
    lajittele(merkkiTaulu, taulukonKoko);
    float liukuTaulu[4] = {3.5, 1.8, 7.13, 6.223};
    taulukonKoko = sizeof(liukuTaulu) / sizeof(liukuTaulu[0]);
    lajittele(liukuTaulu, taulukonKoko);
}
