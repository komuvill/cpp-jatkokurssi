Luokan esittely:

#ifndef TAULUKKO_H
#define TAULUKKO_H
#include <iostream>

using namespace std;

template<typename T>

class taulukko
{
public:
    taulukko(int koko);
    void syotaLuku(int indeksi, T arvo);
private:
    T *taulu;
};

#endif // TAULUKKO_H

Luokan toteutus:

#include "taulukko.h"

template<class T>
taulukko<T>::taulukko(int koko)
{
    taulu = new T[koko];
}

template<class T>
void taulukko<T>::syotaLuku(int indeksi, T arvo)
{
    taulu[indeksi] = arvo;
}

template class taulukko<int>;
template class taulukko<float>;

Esimerkkiohjelma:

#include <iostream>
#include "taulukko.h"

using namespace std;

int main(int argc, char *argv[])
{
    int lkm = 0;
    int valinta = 0;
    int valintaLippu = 0;
    cout << "Montako lukua haluat syöttää?\n";
    cin >> lkm;
    while(valintaLippu == 0){
    cout << "(1) Kokonaislukuja vai (2) liukulukuja?\n";
    cin >> valinta;

    if (valinta == 1){
        valintaLippu++;
        taulukko<int> taulu(lkm);
        for(int i = 0; i < lkm; i++){
            cout << "Syötä " << i+1 <<". alkio" << endl;
            int apu;
            cin >> apu;
            taulu.syotaLuku(i, apu);
        }
    }else if(valinta == 2){
        taulukko<float> taulu(lkm);
        valintaLippu++;
        for(int i = 0; i < lkm; i++){
            cout << "Syötä " << i+1 <<". alkio" << endl;
            float apu;
            cin >> apu;
            taulu.syotaLuku(i, apu);
        }
    }else{
        cout << "Syötetty valinta ei kelpaa\n";
    }
    }
    return 0;

}
